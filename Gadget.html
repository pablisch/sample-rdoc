def delete_app(name)
  app = apps.find { |installed| installed.name == name }
  @apps.delete(app) unless app.nil?
end

def password=(new_password)
  if validate_password(new_password)
    @password = new_password
  else

    puts "Passwords require a minimum of 6 characters and must inlcude a letter and number."
